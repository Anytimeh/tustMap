map.setMarker(position, options)
  • 添加简单标注
    myMap.setMarker([117.278097,39.077101]);
options选项
  • 目前支持4个参数
    var markerOptions = {
        icon: Object, // 标记的图标
        title: String, // 鼠标移动到标记上的文字
        extData: any, // 扩展属性，可以为id
        _click: Function // 点击标记的事件函数
    };
  • myMap.setMarker([117.278097,39.077101], markerOptions);
