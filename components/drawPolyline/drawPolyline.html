options
  • 目前支持7个参数
    var options = {
        strokeColor: String, // 线段颜色，默认值'#3366FF'
        strokeOpacity: Number, // 线段透明度，默认值1
        strokeWeight: Number, // 线宽，默认值5
        strokeStyle: String, // 线段样式，默认值'solid'
        strokeDasharray: Array, // 默认值[10, 5]
        extData: any, // 扩展属性，可以为id，默认值null
        _draw: Function // 线段画完的事件，默认值null
    };
