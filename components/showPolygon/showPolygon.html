options
  • 目前支持7个参数
    var polylineOptions = {
        strokeColor: String, // 边线色，默认值'#FF33FF'
        strokeOpacity: Number, // 边线透明度，默认值0.2
        strokeWeight: Number, // 边线宽，默认值3
        fillColor: String, // 填充色，默认值'#1791fc'
        fillOpacity: Number, // 默认值0.35
        extData: any, // 扩展属性，可以为id
        _click: Function, // 点击多边形的事件函数
        _dblclick: Function // 双击多边形的事件函数
    };
